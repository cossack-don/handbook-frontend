# Массивы

Несколько примеров типизирования массивов.

// Первый способ
// Можно указать string / numbers / boolean
let arr: string[] = ['a', 'b', 'c', 'd', 'e'];

// Второй способ (Дженерики)
let arr: Array<string> = ['a', 'b', 'c', 'd', 'e'];
