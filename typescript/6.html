# Интерфейсы

Типизация объектов в TypeScript являются интерфесы. Они позволяют создавать новые типы данных, описывающие структуру объектов. Интерфесы создаются с помощью ключевого слова interface, после которого идет название интерфейса (пишется с большой буквы)

interface User {
    name: string,
    age: number
}

let user: User = {
	name: 'john',
	 age: 30
 };
